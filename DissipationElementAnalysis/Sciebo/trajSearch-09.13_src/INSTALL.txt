This software is tested with Intel Compiler Suite and IBM Compiler Suite.

1. compile HDF5
---------------
	export CC=icc     
	export CXX=icc      
	export FC=ifort
	export CFLAGS='-O3'
	export CXXFLAGS='-O3'
	export FCFLAGS='-O3'

	./configure \
        --enable-fortran \
        --enable-production \
        --enable-shared=no \
        --enable-static-exec
